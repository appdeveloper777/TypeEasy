    id: int;
    name: string;
    company: string;

    __constructor(id: int, name: string, company: string) {
        this.id = id;
        this.name = name;
        this.company = company;
    }
}

endpoint {
    [HttpGet("/api/dynamic")]
    GetDynamic() {
        let p = new Person(101, "Dynamic User", "TypeEasy Inc");
        return xml(p);
    }

class PersonJson {
      nombre: string;
      company: string;
      edad: int;

      __constructor(_edad: int, _nombre: string, _company: string) {
        this.edad = _edad;
        this.nombre = _nombre;
        this.company = _company;
      }
    }

    endpoint {
      [HttpGet("/api/usuario_json/segundo-perfil")]
      GetUserProfileDos() {
        let usuario = new PersonJson(36, "Fernando Ingunza", "Roussecloud");
        return json(usuario);
      }
    }
